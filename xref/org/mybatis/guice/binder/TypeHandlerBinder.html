13  package org.mybatis.guice.binder;
14  
15  import com.google.inject.TypeLiteral;
16  
17  import org.apache.ibatis.type.TypeHandler;
18  
19  /**
20   * Bind the given {@code TypeHandler} to an already defined type.
21   */
22  public interface TypeHandlerBinder<T> {
23  
24    /**
25     * Bind the given {@code TypeHandler} to an already defined type.
26     *
27     * @param handler The {@code TypeHandler} has to be bound
28     */
29    void with(Class<? extends TypeHandler<? extends T>> handler);
30  
31    /**
32     * Bind the given {@code TypeHandler} to an already defined type.
33     *
34     * @param handler The {@code TypeHandler} has to be bound
35     */
36    void with(TypeLiteral<? extends TypeHandler<? extends T>> handler);
37  
38    /**
39     * Bind the given {@code TypeHandler} to an already defined type.
40     *
41     * @param handler The {@code TypeHandler} has to be bound
42     */
43    void withProvidedTypeHandler(Class<? extends TypeHandler<? extends T>> handler);
44  
45    /**
46     * Bind the given {@code TypeHandler} to an already defined type.
47     *
48     * @param handler The {@code TypeHandler} has to be bound
49     */
50    void withProvidedTypeHandler(TypeLiteral<? extends TypeHandler<? extends T>> handler);
51  
52  }
