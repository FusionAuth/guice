13  package org.mybatis.guice.provision;
14  
15  import com.google.inject.Binder;
16  import com.google.inject.MembersInjector;
17  import com.google.inject.spi.ProvisionListener;
18  
19  import javax.inject.Provider;
20  
21  import org.mybatis.guice.configuration.ConfigurationProvider;
22  import org.mybatis.guice.configuration.settings.ConfigurationSetting;
23  import org.mybatis.guice.configuration.settings.MapperConfigurationSetting;
24  
25  public final class ConfigurationProviderProvisionListener implements ProvisionListener {
26  
27    private final ConfigurationProviderProvisionAction action;
28  
29    ConfigurationProviderProvisionListener(ConfigurationProviderProvisionAction action) {
30      this.action = action;
31    }
32  
33    @Override
34    public <T> void onProvision(ProvisionInvocation<T> provision) {
35      ConfigurationProvider configurationProvider = (ConfigurationProvider) provision.provision();
36      this.action.perform(configurationProvider);
37    }
38  
39    public static <P extends Provider<? extends ConfigurationSetting>> ConfigurationProviderProvisionListener create(
40        final P configurationSettingProvider, final Binder binder) {
41      @SuppressWarnings("unchecked")
42      final MembersInjector<P> membersInjector = (MembersInjector<P>) binder
43          .getMembersInjector(configurationSettingProvider.getClass());
44      return new ConfigurationProviderProvisionListener(new ConfigurationProviderProvisionAction() {
45        @Override
46        public void perform(ConfigurationProvider configurationProvider) {
47          membersInjector.injectMembers(configurationSettingProvider);
48          configurationProvider.addConfigurationSetting(configurationSettingProvider.get());
49        }
50      });
51    }
52  
53    public static ConfigurationProviderProvisionListener create(final ConfigurationSetting configurationSetting) {
54      return new ConfigurationProviderProvisionListener(new ConfigurationProviderProvisionAction() {
55        @Override
56        public void perform(ConfigurationProvider configurationProvider) {
57          configurationProvider.addConfigurationSetting(configurationSetting);
58        }
59      });
60    }
61  
62    public static ConfigurationProviderProvisionListener create(
63        final MapperConfigurationSetting mapperConfigurationSetting) {
64      return new ConfigurationProviderProvisionListener(new ConfigurationProviderProvisionAction() {
65        @Override
66        public void perform(ConfigurationProvider configurationProvider) {
67          configurationProvider.addMapperConfigurationSetting(mapperConfigurationSetting);
68        }
69      });
70    }
71  
72    private static interface ConfigurationProviderProvisionAction {
73      void perform(ConfigurationProvider configurationProvider);
74    }
75  }
