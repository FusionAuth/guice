13  package org.mybatis.guice.multidstest;
14  
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  import javax.naming.InitialContext;
19  import javax.naming.NamingException;
20  
21  public class MockContext extends InitialContext {
22  
23    private static Map<String, Object> bindings = new HashMap<String, Object>();
24  
25    public MockContext(boolean lazy) throws NamingException {
26      super(lazy);
27    }
28  
29    @Override
30    public Object lookup(String name) throws NamingException {
31      return bindings.get(name);
32    }
33  
34    @Override
35    public void bind(String name, Object obj) throws NamingException {
36      bindings.put(name, obj);
37    }
38  }
